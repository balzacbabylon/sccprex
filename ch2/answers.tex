\section{2.1}
    \begin{equation}
        \begin{split}
            \sum_{j=0}^{m-1} 2^{j}  &= 2^{m-1}\sum_{j=0}^{m-1} 2^{-j} \\
                                    &= 2^{m-1}\frac{1-(1/2)^{m-1}}{1-(1/2)} \\
                                    &= 2^{m} - 1 \\      
        \end{split}
    \end{equation}
